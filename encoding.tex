\section{Non-incremental}

\begin{align*}
  &\Pi_h: \varphi^{\text{INIT}} \land \varphi_{h}^{\text{TILS}} \land \varphi_{h}^{\text{DUR\_ACTIONS}} \land \varphi_{h}^{\text{INST\_ACTIONS}} \land \varphi_{h}^{\text{FA}} \land \varphi_{h}^{\text{TIME}} \land \varphi_{h}^{\text{TC}} \land \varphi_{h}^{\text{GOALS}} \land \varphi_{h}^{\text{NSO}} \land \varphi_{h}^{\text{MUTEX}}\\
\end{align*}

\begin{align*}
  &\varphi^{\text{INIT}}: \bigwedge\limits_{f \in F} \left( f_{0} = I(f) \right)
\end{align*}

\begin{align*}
  &\llbracket\tau\rrbracket =
  \begin{cases}
      k, & \text{if } \tau = \text{start} + k\\
      t_{h-1}, & \text{if } \tau = \text{end}
  \end{cases}
\end{align*}

\begin{align*}
  &\varphi_h^{\text{TILS}}:\bigwedge\limits_{[\tau] \; f \coloneqq e \in \text{TILS}} \left( \bigvee\limits_{i \in \{ 1, \ldots, h-1 \}} ( t_i = \llbracket\tau\rrbracket \land f_i = [e]_{i-1}) \right)
\end{align*}

\begin{align*}
  &\varphi_h^{\text{DUR\_ACTIONS}}:\bigwedge\limits_{a \in A} \bigwedge\limits_{s \in \{ 1, \ldots, h \}} ( a_s \implies \varphi_{s, h}^{C_a} \land \varphi_{s, h}^{E_a} \land \varphi_{s, h}^{D_a} ) \\
\end{align*}

\begin{align*}
  &\llbracket\tau\rrbracket_s^a =
  \begin{cases}
      t_s + k, & \text{if } \tau = \text{start} + k\\
      t_s + d_s^a - k, & \text{if } \tau = \text{end} - k
  \end{cases}
\end{align*}

\begin{align*}
  &\!\!\!\!\!\!\!\!\!\varphi_{s, h}^{C_a}: \bigwedge\limits_{[\tau_1, \tau_2] \phi \in C_a} &&\left( \bigwedge\limits_{p \in \{ s, \ldots, h-1 \}} \left( \left( \llbracket\tau_1\rrbracket_s^a \leq t_p < \llbracket\tau_2\rrbracket_s^a \implies [\phi]_p^s \right) \land \left( t_{p-1} < \llbracket\tau_1\rrbracket_s^a \land t_p \geq \llbracket\tau_1\rrbracket_s^a \land \llbracket\tau_1\rrbracket_s^a \leq \llbracket\tau_2\rrbracket_s^a \implies [\phi]_{p-1}^s \right) \right) \right)
\end{align*}

\begin{align*}
  &\varphi_{s, h}^{E_a}: \bigwedge\limits_{[\tau] f \coloneqq e \in E_a} \left( \bigvee\limits_{p \in \{ s, \ldots, h-1 \}} \left( t_p = \llbracket\tau\rrbracket_s^a \land f_{p} = [e]_{p-1}^s \right) \right)
\end{align*}

\begin{align*}
  &\varphi_{s, h}^{D_a}: ([d1]_{s-1}^s \leq d_s^a \leq [d2]_{s-1}^s) \land (t_s + d_s^a \leq t_{h-1})
\end{align*}

\begin{align*}
  & \varphi_{h}^{\text{INST\_ACTIONS}}: \bigwedge\limits_{a \in inst\_A} \bigwedge\limits_{i \in \{1, \ldots, h-1\}} \left( a_i \implies \left( \bigwedge\limits_{f \coloneqq e \in a} \left( f_{i} = [e]_{i-1}^{i} \right) \land \bigwedge\limits_{e \in a} [e]_{i-1}^{i} \right) \right) \\
\end{align*}

\begin{align*}
  &\varphi_h^{\text{FA}}: \bigwedge\limits_{f \in F} \bigwedge\limits_{i \in \{ 1, \ldots, h-1 \}} \left( f_{i-1} \neq f_i \implies \left( \bigvee\limits_{[\tau] f \coloneqq e \in \text{TILS}} t_i = \llbracket\tau\rrbracket \right) \lor \left( \bigvee\limits_{a \in A} \bigvee\limits_{[\tau] f \coloneqq e \in E_a} \bigvee\limits_{j \in \{ 1, \ldots, i \}} a_j \land t_i = \llbracket\tau\rrbracket_j^a \right) \right)
\end{align*}


\begin{align*}
  &\varphi_{h}^{\text{TIME}}: \; t_0 = -1 \; \land \; 0 \leq t_1 \; \land \; \bigwedge\limits_{i \in \{2, \ldots, h-1\}} \left( t_{i-1} < t_i \right)
\end{align*}

\begin{align*}
  &\varphi_{h}^{\text{TC}}: \bigwedge\limits_{f[lb, up] \in F} \bigwedge\limits_{i \in \{0, \ldots, h-1\}} f_i \geq lb \land f_i \leq ub\\
\end{align*}

\begin{align*}
  &\varphi_{h}^{\text{GOALS}}: && \bigwedge\limits_{\substack{[\tau_1, \tau_2] \; e \in G\\\ \tau_1 \ne \text{end} \land \tau_2 \ne \text{end} \\ \tau_1 \leq \tau_2}} &&\left( \bigwedge\limits_{s \in \{1, \ldots, h-1\}} \left(t_s \geq \llbracket \tau_1 \rrbracket \land t_s < \llbracket \tau_2 \rrbracket \right) \implies [e]_{s} \right)\\
  & && && \land \left( \bigwedge\limits_{s \in \{1, \ldots, h-1\}} \left( t_{s-1} < \llbracket \tau_1 \rrbracket \land t_s \geq \llbracket \tau_1 \rrbracket \land \right) \implies [e]_{s-1} \right)\\
  & && && \land \left( \llbracket \tau_2 \rrbracket \leq t_{h-1} \right) \land \left( \left( \llbracket \tau_1 \rrbracket = 0 \right) \implies [e]_0 \right) \\ \\
  & &&\land \bigwedge\limits_{\substack{[\tau, \text{end}] \; e \in G\\\ \tau \ne \text{end}}} &&\left( \bigwedge\limits_{s \in \{1, \ldots, h-1\}} \left( t_s \geq \llbracket \tau \rrbracket \right) \implies [e]_{s} \right)\\
  & && && \land \left( \bigwedge\limits_{s \in \{1, \ldots, h-1\}} \left( t_{s-1} < \llbracket \tau \rrbracket \land t_s \geq \llbracket \tau \rrbracket \right) \implies [e]_{s-1} \right)\\ 
  & && && \land \left( \left( \llbracket \tau \rrbracket = 0 \right) \implies [e]_0 \right) \\ \\
  & &&\land \bigwedge\limits_{[\text{end}] \; e \in G} &&\left( [e]_{h-1} \right)\\
\end{align*}

\begin{align*}
  &\varphi_{h}^{\text{NSO}}: \bigwedge\limits_{a \in A} \bigwedge\limits_{\overrightarrow{p} \in \text{params}(a)} \bigwedge\limits_{i \in \{ 1, \ldots, h-1 \}} \bigwedge\limits_{j \in \{ 1, \ldots, i-1 \}} \left( \left( a_i \land a_j \land \overrightarrow{p(a)}_i = \overrightarrow{p(a)}_j\right) \implies t_j + d_j^a \leq t_i \right)
\end{align*}

\begin{align*}
  &\varphi_{h}^{\text{MUTEX}}:&&\bigwedge\limits_{a \in A \cup \text{inst}_A} \bigwedge\limits_{\xi_1 \equiv [\tau_1, \tau_2]\phi \in a} \bigwedge\limits_{i \in \{1, \ldots, h-1\}} \left( \vphantom{\bigwedge\limits_{a \in A \cup \text{inst}_A} \bigwedge\limits_{\xi_1 \equiv [\tau_1] \text{ev}_1 \in a} \bigwedge\limits_{i \in \{1, \ldots, h-1\}}} \right.\\ & && \bigwedge\limits_{b \in A \cup \text{inst}_A} \bigwedge\limits_{\substack{\xi_2 \equiv [\tau^\prime]f \coloneqq e \in E_b:\\\text{is\_mutex}(\xi_1,\xi_2)}}\bigwedge\limits_{j \in \{1, \ldots, h-1\}} \neg (a_i \land b_j \land (\llbracket \tau_1 \rrbracket_i^a = \llbracket \tau^\prime \rrbracket_j^b \lor \llbracket \tau_2 \rrbracket_i^a = \llbracket \tau^\prime \rrbracket_j^b)) \land \\& && \left. \bigwedge\limits_{\substack{\xi_2 \equiv [\tau_1^\prime, \tau_2^\prime]\phi \in \text{TILS} \cup \text{G}:\\ \text{is\_mutex}(\xi_1, \xi_2)}} \neg (a_i \land (\llbracket \tau_1 \rrbracket_i^a = \llbracket \tau_1^\prime \rrbracket \lor \llbracket \tau_1 \rrbracket_i^a = \llbracket \tau_2^\prime \rrbracket \lor \llbracket \tau_2 \rrbracket_i^a = \llbracket \tau_1^\prime \rrbracket \lor \llbracket \tau_2 \rrbracket_i^a = \llbracket \tau_2^\prime \rrbracket))\vphantom{\bigwedge\limits_{a \in A \cup \text{inst}_A} \bigwedge\limits_{\text{ev}_1 \equiv [\tau_1] e_1 \in a} \bigwedge\limits_{i \in \{1, \ldots, h-1\}}}\right)
\end{align*}

\begin{align*}
  &\varphi_h^{\text{ACTIONS}}:\bigwedge\limits_{a \in A} \bigwedge\limits_{s \in \{ 1, \ldots, h \}} ( a_s \implies \varphi_{s, h}^{C_a} \land \varphi_{s, h}^{E_a} \land \varphi_{s, h}^{D_a} ) \\

\begin{align*}
  &\!\!\!\!\!\!\!\!\!\!\!\varphi_{s, h}^{C_a}: \bigwedge\limits_{[\tau_1, \tau_2] \phi \in C_a} &&\left( \bigwedge\limits_{p \in \{ s, \ldots, h-1 \}} \left( \left( \llbracket\tau_1\rrbracket_s^a \leq t_p < \llbracket\tau_2\rrbracket_s^a \implies [\phi]_p^s \right) \land \left( t_{p-1} < \llbracket\tau_1\rrbracket_s^a \land t_p \geq \llbracket\tau_1\rrbracket_s^a \land \llbracket\tau_1\rrbracket_s^a \leq \llbracket\tau_2\rrbracket_s^a \implies [\phi]_{p-1}^s \right) \right) \land \right.\\

\end{align*}

\begin{align*}
  &\varphi_{s, h}^{E_a}: \bigwedge\limits_{[\tau] f \coloneqq e \in E_a} \left( \bigvee\limits_{p \in \{ s, \ldots, h-1 \}} \left( t_p = \llbracket\tau\rrbracket_s^a \land f_{p} = [e]_{p-1}^s \right) \lor t_{h-1} < \llbracket\tau\rrbracket_s^a \right)
\end{align*}

\begin{align*}
  &\varphi_{s, h}^{D_a}: [d1]_{s-1}^s \leq d_s^a \leq [d2]_{s-1}^s
\end{align*}

\begin{align*}
  &\varphi_h^{\text{FA}}: \bigwedge\limits_{f \in F} \bigwedge\limits_{i \in \{ 1, \ldots, h-1 \}} \left( f_{i-1} \neq f_i \implies \left( \bigvee\limits_{[\tau] f \coloneqq e \in \text{TILS}} t_i = \llbracket\tau\rrbracket \right) \lor \left( \bigvee\limits_{a \in A} \bigvee\limits_{[\tau] f \coloneqq e \in E_a} \bigvee\limits_{j \in \{ 1, \ldots, i \}} a_j \land t_i = \llbracket\tau\rrbracket_j^a \right) \right)

\end{align*}

\begin{align*}

\begin{align*}
  &\varphi_{h}^{\text{GOALS}}: && \bigwedge\limits_{\substack{[\tau_1, \tau_2] \; e \in G\\\ \tau_1 \ne \text{end} \land \tau_2 \ne \text{end} \\ \tau_1 \leq \tau_2 }} &&\left( \bigwedge\limits_{s \in \{1, \ldots, h-1\}} \left(t_s \geq \llbracket \tau_1 \rrbracket \land t_s < \llbracket \tau_2 \rrbracket \right) \implies [e]_{s} \right)\\
  & && && \land \left( \bigwedge\limits_{s \in \{1, \ldots, h-1\}} \left( t_{s-1} < \llbracket \tau_1 \rrbracket \land t_s \geq \llbracket \tau_1 \rrbracket \right) \implies [e]_{s-1} \right)\\
  & && &&  \land \left( \left( \llbracket \tau_1 \rrbracket = 0 \right) \implies [e]_0 \right) \\ 
  & && && \land \left( t_{h-1}< \llbracket \tau_2 \rrbracket \implies \left( [\phi]_{h-1} \lor \bigvee\limits_{f \in \text{vars}(\phi)} \text{mod}_f \right) \right)\\
  & &&\land \bigwedge\limits_{\substack{[\tau, \text{end}] \; e \in G\\\ \tau \ne \text{end}}} &&\left( \bigwedge\limits_{s \in \{1, \ldots, h-1\}} \left( t_s \geq \llbracket \tau \rrbracket \right) \implies [e]_{s} \right)\\
  & && && \land \left( \bigwedge\limits_{s \in \{1, \ldots, h-1\}} \left( t_{s-1} < \llbracket \tau \rrbracket \land t_s \geq \llbracket \tau \rrbracket \right) \implies [e]_{s-1} \right)\\ 
  & && && \land \left( \left( \llbracket \tau \rrbracket = 0 \right) \implies [e]_0 \right) \\
  & && && \land \left( t_{h-1} < \llbracket \tau \rrbracket \implies \left( [\phi]_{h-1} \lor \bigvee\limits_{f \in \text{vars}(\phi)} \text{mod}_f \right) \right)
  \\
  & &&\land \bigwedge\limits_{[\text{end}] \phi \in G} &&\left([\phi]_{h-1} \lor \bigvee\limits_{f \in \text{vars}(\phi)} \text{mod}_f \right)\\
\end{align*}

\begin{align*}
